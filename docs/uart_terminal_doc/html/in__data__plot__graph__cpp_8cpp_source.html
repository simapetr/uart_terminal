553 void data_plot::redraw(void)
554 {
555 uint32_t signal_cnt_ui32;
556 wxString value_text_str;
557 uint32_t cols_num_ui32;
558 uint32_t sample_cnt_ui32;
559 
560  if (!this->l_pause_b)
561  {
562  // Add new variable grid cell
563  while ((uint32_t)this->lp_graph_signal_wxgrid->GetCols() < this->lv_signal_data_plot_buffer.size())
564  {
565  cols_num_ui32 = this->lp_graph_signal_wxgrid->GetCols();
566  this->lp_graph_signal_wxgrid->AppendCols();
567  this->lp_graph_signal_wxgrid->SetColLabelValue(cols_num_ui32, this->lv_signal_data_plot_buffer[cols_num_ui32].data_signal_name_str);
568  }
569  for (signal_cnt_ui32 = 0 ; signal_cnt_ui32 < this->get() ; signal_cnt_ui32++)
570  {
571  if (this->lv_signal_data_plot_buffer[signal_cnt_ui32].p_buffer_measure_data_polygon == NULL)
572  {
573  this->lp_graph_mpwindow->AddLayer( this->lv_signal_data_plot_buffer[signal_cnt_ui32].p_buffer_measure_data_polygon = new mpPolygon(this->lv_signal_data_plot_buffer[signal_cnt_ui32].data_signal_name_str) );
574  this->lv_signal_data_plot_buffer[signal_cnt_ui32].p_buffer_measure_data_polygon->SetPen( this->lv_signal_data_plot_buffer[signal_cnt_ui32].data_signal_style_pen );
575  this->lv_signal_data_plot_buffer[signal_cnt_ui32].p_buffer_measure_data_polygon->ShowName(false);
576  }
577  this->lv_signal_data_plot_buffer[signal_cnt_ui32].p_buffer_measure_data_polygon->setPoints(this->lv_time_d, this->lv_signal_data_plot_buffer[signal_cnt_ui32].v_buffer_data_plot_d, false );
578  // set text value
579  value_text_str.Printf("%f",(float)this->lv_signal_data_plot_buffer[signal_cnt_ui32].v_buffer_data_plot_d.at(this->lv_signal_data_plot_buffer[signal_cnt_ui32].v_buffer_data_plot_d.size() - 1));
580  value_text_str.Replace(wxT(".000000"), wxT(""),true);
581  this->lp_graph_signal_wxgrid->SetCellValue(value_text_str, 0, signal_cnt_ui32);
582  }
583  this->lp_graph_mpwindow->SetPosX(this->lv_time_d[0]);
584  // Add time sample
585  for(sample_cnt_ui32 = 0 ; sample_cnt_ui32 < this->l_sample_cnt_ui32 ; sample_cnt_ui32++)
586  {
587  // Time increment
588  this->l_cnt_d += this->l_step_d;
589  this->lv_time_d.push_back(this->l_cnt_d);
590  this->lv_time_d.erase(this->lv_time_d.begin());
591  }
592  // Clear sample counter
593  this->l_sample_cnt_ui32 = 0;
594  }
595  return;
596 }
